// Endpoint untuk mengambil data (Method: GET)
function doGet(e) {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();

    // --- 1. AMBIL DATA SANTRI ---
    const sheetSantri = ss.getSheetByName("Data Santri");
    const santriList = [];
    if (sheetSantri) {
      const dataRows = sheetSantri.getDataRange().getValues();
      for (let i = 1; i < dataRows.length; i++) {
        if (dataRows[i][0] || dataRows[i][1]) {
        santriList.push({
                id: i,
                nis: dataRows[i][0].toString().trim(),
                nama: dataRows[i][1].toString().trim(),
                kelas: dataRows[i][2].toString().trim(),
                status: dataRows[i][3] ? dataRows[i][3].toString().trim() : "Belum Bayar"
              });
        }
      }
    }

    // --- 2. AMBIL DATA PEMBAYARAN ---
    const sheetPembayaran = ss.getSheetByName("Pembayaran");
    const pembayaranList = [];
    let totalMasuk = 0;
    
    if (sheetPembayaran) {
      const bayarRows = sheetPembayaran.getDataRange().getValues();
      for (let i = 1; i < bayarRows.length; i++) {
        if (bayarRows[i][1]) {
          const nominal = Number(bayarRows[i][4]) || 0;
       pembayaranList.push({
                row: i + 1, // TAMPILKAN BARIS ASLI GOOGLE SHEETS (Array index + 1)
                tanggal: bayarRows[i][0],
                nama: bayarRows[i][1].toString().trim(),
                kelas: bayarRows[i][2].toString().trim(),
                jenis: bayarRows[i][3].toString().trim(),
                nominal: nominal
              });
          totalMasuk += nominal;
        }
      }
    }

    // --- 3. AMBIL PENGATURAN BIAYA DARI TAB "Pengaturan" ---
    const sheetSetting = ss.getSheetByName("Pengaturan");
    let settingBiaya = {
      tk_ujian: 50000, tk_rapor: 40000,
      ibt_ujian: 100000, ibt_rapor: 75000,
      sana_ujian: 150000, sana_rapor: 100000
    };
    if (sheetSetting) {
      const dataSetting = sheetSetting.getDataRange().getValues();
      for (let i = 1; i < dataSetting.length; i++) {
        const key = dataSetting[i][0];
        const val = Number(dataSetting[i][1]);
        if (key) settingBiaya[key] = val;
      }
    }

    // --- 4. AMBIL DATA PENGELUARAN ---
    const sheetPengeluaran = ss.getSheetByName("Pengeluaran");
    let totalKeluarUjian = 0;
    let totalKeluarRapor = 0;
    
    if (sheetPengeluaran) {
      const keluarRows = sheetPengeluaran.getDataRange().getValues();
      for (let i = 1; i < keluarRows.length; i++) {
        if (keluarRows[i][1]) {
          const kategori = keluarRows[i][1].toString().trim();
          const nominal = Number(keluarRows[i][2]) || 0;
          
          if (kategori === "Uang Ujian") totalKeluarUjian += nominal;
          if (kategori === "Uang Rapor") totalKeluarRapor += nominal;
        }
      }
    }

    // --- 5. SIAPKAN PAYLOAD UNTUK FRONTEND ---
    const data = {
      statistik: {
        totalSantri: santriList.length,
        pemasukanUjian: pembayaranList.filter(p => p.jenis === 'Uang Ujian').reduce((sum, p) => sum + p.nominal, 0),
        pemasukanRapor: pembayaranList.filter(p => p.jenis === 'Uang Rapor').reduce((sum, p) => sum + p.nominal, 0),
        
        pengeluaranUjian: totalKeluarUjian,
        pengeluaranRapor: totalKeluarRapor,
        
        targetUjian: santriList.reduce((sum, s) => {
            let t = s.kelas.includes('TK') ? settingBiaya.tk_ujian : (s.kelas.includes('IBT') ? settingBiaya.ibt_ujian : settingBiaya.sana_ujian);
            return sum + t;
        }, 0),
        targetRapor: santriList.reduce((sum, s) => {
            let t = s.kelas.includes('TK') ? settingBiaya.tk_rapor : (s.kelas.includes('IBT') ? settingBiaya.ibt_rapor : settingBiaya.sana_rapor);
            return sum + t;
        }, 0)
      },
      santri: santriList,
      pembayaran: pembayaranList,
      setting: settingBiaya
    };
    
    return ContentService.createTextOutput(JSON.stringify(data))
      .setMimeType(ContentService.MimeType.JSON);

  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({ error: error.toString() })).setMimeType(ContentService.MimeType.JSON);
  }
}

// Endpoint untuk menyimpan data (Method: POST)
function doPost(e) {
  try {
    // Validasi pencegah error CORS
    if (!e || !e.postData || !e.postData.contents) {
      return ContentService.createTextOutput(JSON.stringify({ success: false, message: "Data kosong dikirim ke server." })).setMimeType(ContentService.MimeType.JSON);
    }

    const request = JSON.parse(e.postData.contents);
    const ss = SpreadsheetApp.getActiveSpreadsheet();

    // --- AKSI: IMPORT SANTRI ---
    if (request.action === 'import_santri') {
      const sheet = ss.getSheetByName("Data Santri");
      const dataBaru = request.data;
      const barisBaru = dataBaru.map(santri => [santri.NIS, santri.Nama, santri.Kelas, "Belum Bayar"]);
      sheet.getRange(sheet.getLastRow() + 1, 1, barisBaru.length, barisBaru[0].length).setValues(barisBaru);
      return ContentService.createTextOutput(JSON.stringify({ success: true, message: `${barisBaru.length} data santri berhasil diimport!` })).setMimeType(ContentService.MimeType.JSON);
    }

    // --- AKSI: HAPUS PEMBAYARAN ---
    if (request.action === 'hapus_pembayaran') {
      const sheet = ss.getSheetByName("Pembayaran");
      sheet.deleteRow(request.data.index);
      return ContentService.createTextOutput(JSON.stringify({ success: true, message: "Data berhasil dihapus!" })).setMimeType(ContentService.MimeType.JSON);
    }
    
    // --- AKSI: SIMPAN PEMBAYARAN (SISTEM CICILAN) ---
    if (request.action === 'simpan_pembayaran') {
      let sheetPembayaran = ss.getSheetByName("Pembayaran");
      if (!sheetPembayaran) {
        sheetPembayaran = ss.insertSheet("Pembayaran");
        sheetPembayaran.appendRow(["Tanggal", "Nama Santri", "Kelas", "Jenis Pembayaran", "Nominal"]);
        sheetPembayaran.getRange("A1:E1").setFontWeight("bold");
      }
      
      // Ambil Pengaturan Target
      let sheetSetting = ss.getSheetByName("Pengaturan");
      let settingBiaya = { tk_ujian: 50000, tk_rapor: 40000, ibt_ujian: 100000, ibt_rapor: 75000, sana_ujian: 150000, sana_rapor: 100000 };
      if (sheetSetting) {
        let dataSetting = sheetSetting.getDataRange().getValues();
        for (let i = 1; i < dataSetting.length; i++) {
          if (dataSetting[i][0]) settingBiaya[dataSetting[i][0]] = Number(dataSetting[i][1]);
        }
      }

      let namaInput = (request.data.nama || "").toString().trim();
      let jenisInput = (request.data.jenis || "").toString().trim();
      let kelasInput = (request.data.kelas || "").toString().trim(); 
      let kelasTeks = kelasInput.toUpperCase();
      let nominalInput = Number(request.data.nominal) || 0;
      
      // Tentukan target santri ini
      let target = 0;
      if (jenisInput === "Uang Ujian") {
        if (kelasTeks.includes("TK")) target = settingBiaya.tk_ujian;
        else if (kelasTeks.includes("IBT")) target = settingBiaya.ibt_ujian;
        else if (kelasTeks.includes("SANA")) target = settingBiaya.sana_ujian;
      } else {
        if (kelasTeks.includes("TK")) target = settingBiaya.tk_rapor;
        else if (kelasTeks.includes("IBT")) target = settingBiaya.ibt_rapor;
        else if (kelasTeks.includes("SANA")) target = settingBiaya.sana_rapor;
      }

      // Hitung total cicilan di server
      const dataTransaksi = sheetPembayaran.getDataRange().getValues();
      let totalTerbayar = 0;
      for (let i = 1; i < dataTransaksi.length; i++) {
        let namaDiSheet = dataTransaksi[i][1] ? dataTransaksi[i][1].toString().trim() : "";
        let jenisDiSheet = dataTransaksi[i][3] ? dataTransaksi[i][3].toString().trim() : "";
        if (namaDiSheet === namaInput && jenisDiSheet === jenisInput) {
          totalTerbayar += (Number(dataTransaksi[i][4]) || 0);
        }
      }

      // Tolak jika kelebihan bayar 
      if (target > 0) {
        if (totalTerbayar >= target) {
          return ContentService.createTextOutput(JSON.stringify({ success: false, message: `DITOLAK SERVER: Sudah LUNAS!` })).setMimeType(ContentService.MimeType.JSON);
        }
        if ((totalTerbayar + nominalInput) > target) {
          return ContentService.createTextOutput(JSON.stringify({ success: false, message: `DITOLAK SERVER: Nominal terlalu besar!` })).setMimeType(ContentService.MimeType.JSON);
        }
      }

      // Simpan jika aman
      sheetPembayaran.appendRow([new Date(), namaInput, kelasInput, jenisInput, nominalInput]);
      
      // Update status di tab "Data Santri"
      let sheetSantri = ss.getSheetByName("Data Santri");
      if (sheetSantri) {
        let dataSantri = sheetSantri.getDataRange().getValues();
        for (let i = 1; i < dataSantri.length; i++) {
          if (dataSantri[i][1] === namaInput) { 
            sheetSantri.getRange(i + 1, 4).setValue("Ada Pembayaran");
            break; 
          }
        }
      }

      return ContentService.createTextOutput(JSON.stringify({ success: true, message: "Pembayaran berhasil dicatat!" })).setMimeType(ContentService.MimeType.JSON);
    }

    // --- AKSI: SIMPAN SETTING BIAYA ---
    if (request.action === 'simpan_setting') {
      let sheetSetting = ss.getSheetByName("Pengaturan");
      if (!sheetSetting) sheetSetting = ss.insertSheet("Pengaturan");
      
      sheetSetting.clear();
      const set = request.data;
      const rows = [
        ["Key", "Value"],
        ["tk_ujian", set.tk_ujian], 
        ["tk_rapor", set.tk_rapor],
        ["ibt_ujian", set.ibt_ujian], 
        ["ibt_rapor", set.ibt_rapor],
        ["sana_ujian", set.sana_ujian], 
        ["sana_rapor", set.sana_rapor]
      ];
      sheetSetting.getRange(1, 1, rows.length, 2).setValues(rows);
      return ContentService.createTextOutput(JSON.stringify({ success: true, message: "Pengaturan berhasil dicatat!" })).setMimeType(ContentService.MimeType.JSON);
    }
// --- AKSI BARU: SIMPAN PENGELUARAN ---
    if (request.action === 'simpan_pengeluaran') {
      let sheetPengeluaran = ss.getSheetByName("Pengeluaran");
      
      // Buat tab otomatis jika belum ada
      if (!sheetPengeluaran) {
        sheetPengeluaran = ss.insertSheet("Pengeluaran");
        sheetPengeluaran.appendRow(["Tanggal", "Kategori", "Nominal", "Keterangan"]);
        sheetPengeluaran.getRange("A1:D1").setFontWeight("bold");
      }
      
      let kategori = (request.data.kategori || "").toString().trim();
      let nominal = Number(request.data.nominal) || 0;
      let keterangan = (request.data.keterangan || "").toString().trim();
      
      // ==========================================
      // PENAMBAHAN KEAMANAN: VALIDASI SALDO DI BACKEND
      // ==========================================
      // 1. Hitung Total Pemasukan untuk Kategori (Kas) ini
      let sheetPembayaran = ss.getSheetByName("Pembayaran");
      let totalPemasukan = 0;
      if (sheetPembayaran) {
        let dataMasuk = sheetPembayaran.getDataRange().getValues();
        for (let i = 1; i < dataMasuk.length; i++) {
          let jenisMasuk = dataMasuk[i][3] ? dataMasuk[i][3].toString().trim() : "";
          if (jenisMasuk === kategori) {
            totalPemasukan += (Number(dataMasuk[i][4]) || 0);
          }
        }
      }

      // 2. Hitung Total Pengeluaran Terdahulu untuk Kategori ini
      let totalPengeluaranTerdahulu = 0;
      let dataKeluar = sheetPengeluaran.getDataRange().getValues();
      for (let i = 1; i < dataKeluar.length; i++) {
        let jenisKeluar = dataKeluar[i][1] ? dataKeluar[i][1].toString().trim() : "";
        if (jenisKeluar === kategori) {
          totalPengeluaranTerdahulu += (Number(dataKeluar[i][2]) || 0);
        }
      }

      // 3. Kalkulasi Saldo Aktual dan Blokir jika Minus
      let saldoAktual = totalPemasukan - totalPengeluaranTerdahulu;
      if (nominal > saldoAktual) {
        return ContentService.createTextOutput(JSON.stringify({ 
          success: false, 
          message: `DITOLAK SERVER: Saldo Kas ${kategori} tidak mencukupi! Sisa saldo aktual: Rp ${saldoAktual}` 
        })).setMimeType(ContentService.MimeType.JSON);
      }
      // ==========================================

      // Jika aman, simpan data pengeluaran
      sheetPengeluaran.appendRow([new Date(), kategori, nominal, keterangan]);
      
      return ContentService.createTextOutput(JSON.stringify({ success: true, message: "Pengeluaran berhasil dicatat!" })).setMimeType(ContentService.MimeType.JSON);
    }

    return ContentService.createTextOutput(JSON.stringify({ success: false, message: "Aksi tidak dikenali." })).setMimeType(ContentService.MimeType.JSON);
  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({ success: false, message: error.toString() })).setMimeType(ContentService.MimeType.JSON);
  }
}
